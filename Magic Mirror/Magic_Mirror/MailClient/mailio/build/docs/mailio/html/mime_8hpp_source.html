31 /*
32 To parse elements of a mime part the following alphabet is used as described in [rfc 5322]:
33 ```
34 atext = ALPHA / DIGIT / "!" / "#" / "$" / "%" / "&" / "'" / "*" / "+" / "-" / "/" / "=" / "?" / "~" / "_" / "`" / "{" / "|" / "}" / "~"
35 atom = [CFWS] 1 * atext [CFWS]
36 dot-atom-text = 1 * atext * ("." 1 * atext)
37 dot-atom = [CFWS] dot-atom-text [CFWS]
38 quoted-pair = ("\SLASH" (VCHAR / WSP))
39 comment = "(" *([FWS] ctext / quoted-pair) [FWS] ")"
40 CFWS = (1 * ([FWS] comment) [FWS]) / FWS
41 qtext = \%d33 / \%d35-\%d91 / \%d93-\%d126 ; printable ascii without backslash and quote
42 qcontent = qtext / quoted-pair
43 quoted-string = [CFWS] DQUOTE * ([FWS] qcontent) [FWS] DQUOTE [CFWS]
44 word = atom / quoted-string
45 phrase = 1 * word
46 dtext = \%d33-\%d90 / \%d94-\%d126 ; printable ascii not including brackets and backslash
47 address = [phrase] angle-addr / addr-spec
48 addr-spec = (dot-atom / quoted-string) "@" (dot-atom / dtext)
49 ```
50 */
